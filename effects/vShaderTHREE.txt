// GLSL version
#version 330 core
layout (location = 0) in vec3 aPos;
layout (location = 1) in vec3 aNormal;
layout (location = 2) in vec2 aTexCoord;

// uniforms
uniform mat4 viewMatrix; //matrixModelView;
uniform mat4 modelMatrix;
uniform mat4 projMatrix;

// varyings (output)
out vec3 esVertex, esNormal;
out vec2 texCoord0;

void main()
{
    gl_Position = projMatrix * viewMatrix * modelMatrix * vec4(aPos, 1.0);
    esVertex = vec3(modelMatrix * viewMatrix * vec4(aPos, 1.0));;
    esNormal = aNormal;
    texCoord0 = aTexCoord;
}
